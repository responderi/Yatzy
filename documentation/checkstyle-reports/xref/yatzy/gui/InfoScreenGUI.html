19      public InfoScreenGUI() {
20      }

22      @Override
23      public void run() {
24          frame = new JFrame("Yatzy - How to play");
25          frame.setPreferredSize(new Dimension(590, 210));
26  
27          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
28  
29          createComponents(frame.getContentPane());
30  
31          frame.pack();
32          frame.setVisible(true);
33      }
34  
35      private void createComponents(Container container) {
36          container.add(infoMenu());
37      }
38  
39      private JPanel infoMenu() {
40          JPanel panel = new JPanel(new GridLayout(2, 1));
41          JButton infoButton = new JButton("Understood!");
42          JTextArea infoText = new JTextArea("How to play:\n You will start your turn by pressing 'Reroll' button.\n If you want to reroll some of the dice, reroll them by pressing specific die buttons and then 'Done rolling'.\n If you havent chosen to score earlier, you will be forced to score after two reroll times.\n Scores and player in turn (left to 'Player 1' text) will update always after one's first reroll-button push.");
43          InfoListener infoListener = new InfoListener(frame);
44          infoButton.addActionListener(infoListener);
45          infoText.setEnabled(false);
46          panel.add(infoText);
47          panel.add(infoButton);
48          return panel;
49      }
50  
51      public JFrame getFrame() {
52          return frame;
53      }
