51      @Override
52      public void actionPerformed(ActionEvent ae) {
53          if (game.checkRerolls() == 0) {
54              die1.setText("" + dice.get(0).getValue());
55              die2.setText("" + dice.get(1).getValue());
56              die3.setText("" + dice.get(2).getValue());
57              die4.setText("" + dice.get(3).getValue());
58              die5.setText("" + dice.get(4).getValue());
59              game.addReroll();
60          } else {
61              die1.setText("" + dice.get(0).getValue());
62              die2.setText("" + dice.get(1).getValue());
63              die3.setText("" + dice.get(2).getValue());
64              die4.setText("" + dice.get(3).getValue());
65              die5.setText("" + dice.get(4).getValue());
66              game.resetRerolls();
67              die1Button.setEnabled(false);
68              die2Button.setEnabled(false);
69              die3Button.setEnabled(false);
70              die4Button.setEnabled(false);
71              die5Button.setEnabled(false);
72              doneRolling.setEnabled(false);
73              scoreButton.setEnabled(false);
74              ScoringScreenGUI scoringGui = new ScoringScreenGUI(game, score, player1, player2, dice);
75              SwingUtilities.invokeLater(scoringGui);
76              rerollButton.setEnabled(true);
77          }
78      }
